  filled: number;
  status: 'open' | 'filling' | 'full';
}
 
interface TransitSchedule {
  route: string;
  line: string;
  eta: number; // minutes
  status: 'on-time' | 'delayed' | 'boarding';
}

        return { ...p, filled: newFilled, status };

        let newEta = s.eta - 1;
        let status = s.status;
